  const { email } = req.body;

  // Basic validation
  if (!email || typeof email !== 'string' || !email.includes('@')) {
    return res.status(400).json({ message: 'Valid email is required.' });

"use client";
import React, { useState } from "react";

// Server action must be defined in a separate file or inline below
async function submitWaitlist(formData) {
  "use server";
  const email = formData.get("email");
  try {
    const res = await fetch(`${process.env.NEXT_PUBLIC_API_URL}/waitlist`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ email }),
      cache: "no-store",
    });

    const data = await res.json();

    if (res.status === 201) {
      return { status: "success", message: "You're on the list! We'll be in touch." };
    } else if (res.status === 200 && data.message === "Already on the list!") {
      return { status: "info", message: "You're already on the list!" };
    } else {
      return { status: "error", message: "Something went wrong. Please try again." };
    }
  } catch (e) {
    return { status: "error", message: "Something went wrong. Please try again." };
  }
}

export default function Waitlist() {
  const [email, setEmail] = useState("");
  const [status, setStatus] = useState(null);

  async function handleSubmit(e) {
    e.preventDefault();
    setStatus(null);

    // Use FormData to match server action signature
    const formData = new FormData();
    formData.set("email", email);

    // Call server action
    const result = await submitWaitlist(formData);
    setStatus(result);
  }

  return (
    <form onSubmit={handleSubmit} className="waitlist-form">
      <label htmlFor="waitlist-email" className="waitlist-label">
        Join the waitlist
      </label>
      <input
        id="waitlist-email"
        name="email"
        type="email"
        className="waitlist-input"
        placeholder="you@email.com"
        value={email}
        onChange={(e) => setEmail(e.target.value)}
        required
      />
      <button type="submit" className="waitlist-button">
        Notify me
      </button>
      {status && (
        <div
          className={
            status.status === "success"
              ? "waitlist-success"
              : status.status === "info"
              ? "waitlist-info"
              : "waitlist-error"
          }
        >
          {status.message}
        </div>
      )}
    </form>
  );
}
