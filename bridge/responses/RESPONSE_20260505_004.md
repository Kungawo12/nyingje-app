// apps/mobile/hooks/useAuth.js

import React, { createContext, useContext, useState, useEffect } from 'react';
import AsyncStorage from '@react-native-async-storage/async-storage';
import api from '../lib/api'; // <-- Use the shared axios instance

const AuthContext = createContext();

export function AuthProvider({ children }) {
  const [user, setUser] = useState(null);
  const [token, setToken] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    // Load user and token from AsyncStorage on mount
    const load = async () => {
      const storedUser = await AsyncStorage.getItem('user');
      const storedToken = await AsyncStorage.getItem('token');
      if (storedUser && storedToken) {
        setUser(JSON.parse(storedUser));
        setToken(storedToken);
      }
      setLoading(false);
    };
    load();
  }, []);

  const login = async (user, token) => {
    setUser(user);
    setToken(token);
    await AsyncStorage.setItem('user', JSON.stringify(user));
    await AsyncStorage.setItem('token', token);
  };

  const logout = async () => {
    setUser(null);
    setToken(null);
    await AsyncStorage.removeItem('user');
    await AsyncStorage.removeItem('token');
  };

  return (
    <AuthContext.Provider value={{ user, token, login, logout, loading }}>
      {children}
    </AuthContext.Provider>
  );
}

export function useAuth() {
  return useContext(AuthContext);
}

export { api }; // Export the shared axios instance

// apps/mobile/lib/api.js

import axios from 'axios';
import AsyncStorage from '@react-native-async-storage/async-storage';
import Constants from 'expo-constants';

const baseURL =
  (Constants?.expoConfig?.extra?.apiUrl) ||
  process.env.EXPO_PUBLIC_API_URL ||
  'http://localhost:3002';

const api = axios.create({
  baseURL,
});

// Attach token from AsyncStorage to every request if present
api.interceptors.request.use(
  async (config) => {
    try {
      const token = await AsyncStorage.getItem('token');
      if (token) {
        config.headers = {
          ...config.headers,
          Authorization: `Bearer ${token}`,
        };
      }
    } catch (e) {
      // If AsyncStorage fails, continue without token
    }
    return config;
  },
  (error) => Promise.reject(error)
);

export default api;

import React, { useState } from 'react';
import { View, Text, TextInput, Button, StyleSheet, Alert } from 'react-native';
import { useAuth } from '../../hooks/useAuth';
import api from '../../lib/api';

export default function LoginScreen({ navigation }) {
  const { login } = useAuth();
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [submitting, setSubmitting] = useState(false);
  const [error, setError] = useState('');

  const handleLogin = async () => {
    setSubmitting(true);
    setError('');
    try {
      const res = await api.post('/auth/login', { email, password });
      await login(res.data.user, res.data.token);
    } catch (err) {
      let msg = 'Login failed. Please try again.';
      if (err.response?.data?.error) {
        msg = err.response.data.error;
      } else if (err.response?.data?.message) {
        msg = err.response.data.message;
      }
      setError(msg);
      Alert.alert('Login Error', msg);
    } finally {
      setSubmitting(false);
    }
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Sign In</Text>
      {error ? <Text style={styles.error}>{error}</Text> : null}
      <TextInput
        style={styles.input}
        placeholder="Email"
        autoCapitalize="none"
        autoComplete="email"
        keyboardType="email-address"
        value={email}
        onChangeText={setEmail}
        editable={!submitting}
      />
      <TextInput
        style={styles.input}
        placeholder="Password"
        secureTextEntry
        autoCapitalize="none"
        value={password}
        onChangeText={setPassword}
        editable={!submitting}
      />

}

import React, { useState, useRef } from 'react';
import { View, Text, TextInput, Button, FlatList, StyleSheet, KeyboardAvoidingView, Platform } from 'react-native';
import { useAuth } from '../../hooks/useAuth';
import api from '../../lib/api';

const initialMessages = [
  {
    id: 'welcome',
    role: 'assistant',
    text: "Hello! I'm Nyingje, your compassionate companion. How can I support you today?",
  },
];

export default function ChatScreen() {
  const { user } = useAuth();
  const [messages, setMessages] = useState(initialMessages);
  const [text, setText] = useState('');
  const [sending, setSending] = useState(false);
  const [conversationId, setConversationId] = useState(null);
  const flatListRef = useRef();

  const sendMessage = async () => {
    if (!text.trim()) return;
    const userMsg = {
      id: `${Date.now()}`,
      role: 'user',
      text,
    };
    setMessages((msgs) => [...msgs, userMsg]);
    setText('');
    setSending(true);

    try {
      const res = await api.post('/chat', {
        message: userMsg.text,
        conversationId: conversationId || undefined,
      });
      setConversationId(res.data.conversationId);

      setMessages((msgs) => [
        ...msgs,
        {
          id: `${Date.now()}-reply`,
          role: 'assistant',
          text: res.data.reply,
        },
      ]);
    } catch (err) {
      let msg = 'Sorry, there was a problem connecting to Nyingje.';
      if (err.response?.data?.error) {
        msg = err.response.data.error;
      } else if (err.response?.data?.message) {
        msg = err.response.data.message;
      }
      setMessages((msgs) => [
        ...msgs,
        {
          id: `${Date.now()}-error`,
          role: 'error',
          text: msg,
        },
      ]);
    } finally {
      setSending(false);
      setTimeout(() => {
        flatListRef.current?.scrollToEnd({ animated: true });
      }, 100);
    }
  };

  const renderItem = ({ item }) => {
    let bubbleStyle = styles.bubbleAssistant;
    let textStyle = styles.textAssistant;
    if (item.role === 'user') {
      bubbleStyle = styles.bubbleUser;
      textStyle = styles.textUser;
    } else if (item.role === 'error') {
      bubbleStyle = styles.bubbleError;
      textStyle = styles.textError;
    }
    return (
      <View style={[styles.bubble, bubbleStyle]}>
        <Text style={textStyle}>{item.text}</Text>
      </View>
    );
  };

  return (
    <KeyboardAvoidingView
      style={{ flex: 1, backgroundColor: '#f8f8f8' }}
      behavior={Platform.OS === 'ios' ? 'padding' : undefined}
      keyboardVerticalOffset={80}
    >
      <FlatList
        ref={flatListRef}
        data={messages}
        renderItem={renderItem}
        keyExtractor={(item) => item.id}
        contentContainerStyle={{ padding: 16, paddingBottom: 80 }}
        onContentSizeChange={() => flatListRef.current?.scrollToEnd({ animated: true })}
      />
      <View style={styles.inputRow}>
        <TextInput
          style={styles.input}
          placeholder="Type your message..."
          value={text}
          onChangeText={setText}
          editable={!sending}
          onSubmitEditing={sendMessage}
          returnKeyType="send"
        />
        <Button title={sending ? '...' : 'Send'} onPress={sendMessage} disabled={sending || !text.trim()} />
      </View>
    </KeyboardAvoidingView>
  );
}

const styles = StyleSheet.create({
